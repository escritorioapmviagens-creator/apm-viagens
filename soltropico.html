⚠️ As chamadas SOAP não podem ser feitas diretamente do browser por restrições de CORS. É necessário uma Edge Function no Supabase como proxy. Consulte a secção de configuração abaixo.
📋 Edge Function Supabase (soltropico-proxy)

Crie esta Edge Function no Supabase para servir de proxy SOAP:

// supabase/functions/soltropico-proxy/index.ts import { serve } from "https://deno.land/std@0.168.0/http/server.ts" const ENDPOINT = "https://www3.optitravel.net/optitravel/export/wbs/pkt/wbs_server.php" serve(async (req) => { const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", } if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }) } try { const { soapAction, soapBody } = await req.json() const res = await fetch(ENDPOINT, { method: "POST", headers: { "Content-Type": "text/xml; charset=utf-8", "SOAPAction": soapAction || "", }, body: soapBody, }) const xml = await res.text() return new Response(JSON.stringify({ xml }), { headers: { ...corsHeaders, "Content-Type": "application/json" }, }) } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" }, }) } })
